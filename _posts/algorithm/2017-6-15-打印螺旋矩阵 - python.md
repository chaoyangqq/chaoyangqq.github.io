def spiral_simple(L): 
    result = [[0]*L for n in range(L)]
    row =0   
    col = 0  
    value = 1
    direction = 'r' 
    circle = 1 
    while True:
            if direction == 'r':
                    result[row][col] = value
                    if col>=L-circle:
                            direction = 'd'
                            continue
                    col += 1
            if direction =='d':
                    result[row][col] = value
                    if row >= L-circle:
                            direction = 'l'
                            continue
                    row += 1
            if direction == 'l':
                    result[row][col] = value
                    if col <= circle -1:
                            direction = 'u'
                            continue
                    col -= 1
            if direction == 'u':
                    result[row][col] = value
                    if row-1 <= circle:
                            direction = 'r'
                            circle += 1
                            #continue
                    row -= 1
            value += 1
            if value > L * L :
                    break
    for R in result:

### 1.中学生版

``` stylus
import itertools  
def spiral(n,m):  
    _status = itertools.cycle(['right','down','left','up']) 
    _movemap = {  
        'right':(1,0),  
        'down':(0,1),  
        'left':(-1,0),  
        'up':(0,-1),  
    }  
    pos2no = dict.fromkeys([(x,y) for x in range(n) for y in range(m)])  
    _pos = (0,0)  
    _st = next(_status)  
    for i in range(1,n*m+1):  
        _oldpos = _pos  
        _pos = tuple(map(sum,zip(_pos,_movemap[_st])))  
        if (_pos not in pos2no) or (pos2no[_pos]):
            _st = next(_status)  
            _pos = tuple(map(sum,zip(_oldpos,_movemap[_st])))  
        pos2no[_oldpos] = i  
    return pos2no  
  
def display_spiral(n,m):  
    pos2no = spiral(n,m)  
    for i in range(m):  
        for j in range(n):  
            print pos2no[(j,i)],'\t',  
        print '\n'  
    print '-'*30  
```
